"use strict";
var sidebar = require("../../actions/containers/sidebar");
var initialState = {
    isCollapsed: false
};
function collapsableSidebarReducer(state, action) {
    if (state === void 0) { state = initialState; }
    switch (action.type) {
        case sidebar.ActionTypes.TOGGLE_SIDEBAR:
            return {
                isCollapsed: !state.isCollapsed
            };
        default:
            return state;
    }
}
exports.collapsableSidebarReducer = collapsableSidebarReducer;
;
