"use strict";
var application = require("../../actions/containers/application");
var initialState = {
    isBusy: false
};
function applicationReducer(state, action) {
    if (state === void 0) { state = initialState; }
    switch (action.type) {
        case application.ActionTypes.ENABLE_BUSY_FLAG:
            return {
                isBusy: true
            };
        case application.ActionTypes.DISABLE_BUSY_FLAG:
            return {
                isBusy: false
            };
        default:
            return state;
    }
}
exports.applicationReducer = applicationReducer;
;
